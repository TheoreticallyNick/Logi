### Installing Python3.7 ###
sudo apt-get update && apt-get remove python3
rm -r /usr/bin/python3*
rm -r /usr/local/lib/python3*
rm -r /usr/lib/python3*
sudo apt install build-essential zlib1g-dev libncurses5-dev libgdbm-dev libnss3-dev libssl-dev libreadline-dev libffi-dev wget
curl -O https://www.python.org/ftp/python/3.7.3/Python-3.7.3.tar.xz
tar -xf Python-3.7.3.tar.xz
cd Python-3.7.3
./configure
make
make install

### Dependencies ###
python3 -m pip install smbus pyserial serial Adafruit_BBIO AWSIoTPythonSDK hologram-python xmodem
